; $NOMOD51
; $INCLUDE (8051.MCU)

Q BIT P3.2
U BIT P1.1; 20H.3
V BIT P1.3
W BIT P1.5 ;22H.0
X BIT P2.0 ; 28H.2;
Z BIT P2.5
A1 BIT P1.0; 28H.5
B1 BIT  P1.4 ; 21H.4
C1 BIT P1.6; 25H.0

ORG 0H

MAIN:
   MOV C, W
   ANL C, V
   MOV F0, C
   MOV C, X
   ANL C, /Z
   ORL C, F0
   MOV F0, C
   MOV C, A1
   ORL C, C1
   MOV F1, C
   MOV C, U
   ORL C, B1
   ANL C, F1
   ORL C, F0
   MOV Q, C
END
